const getPosition = (
  x: number,
  y: number,
  boardWidth: number,
  boardHeight: number
) => {
  if (x === 0 && y === 0) {
    return "topLeft";
  } else if (y === 0 && x === boardWidth - 1) {
    return "topRight";
  } else if (y === 0 && x < boardWidth - 1 && x > 0) {
    return "topMiddle";
  } else if (x === 0 && y < boardHeight - 1 && y > 0) {
    return "leftMiddle";
  } else if (x < boardWidth - 1 && x > 0 && y < boardHeight - 1 && y > 0) {
    return "Middle";
  } else if (x === boardWidth - 1 && y < boardHeight - 1 && y > 0) {
    return "rightMiddle";
  } else if (y === boardHeight - 1 && x < boardWidth - 1 && x > 0) {
    return "bottomMiddle";
  } else if (x === 0 && y === boardHeight - 1) {
    return "bottomLeft";

    return "bottomRight";
  }
};
export default getPosition;
